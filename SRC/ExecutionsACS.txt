#library "ExecutionsACS"
#include "zcommon.acs"

script "Executions_GetTargetMaxHealth" (void)
{
	int enemySpawnHealth = GetActorProperty(0, APROP_SpawnHealth);
	SetResultValue(enemySpawnHealth);
}


script "Executions_GetTargetCurrentHealth" (void)
{ 
	int enemyHealth = GetActorProperty(0, APROP_Health);
	SetResultValue(enemyHealth);
}

//ZScript was being a fucking bitch so I decided to make this into an ACS Script.
//If y'all want to convert this is a ZScript EventHandler, go ahead. -JM
script "Executions_NewLevelCheck" ENTER
{
	TakeInventory("ExecutionToken", 1);
	SetPlayerProperty(TRUE, OFF, PROP_TOTALLYFROZEN);
	SetActorFlag(0, "NODAMAGE", False);
}
